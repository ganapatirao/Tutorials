1. C Fundamentals

Q1: What is C and what are its key features?

C is a general-purpose, procedural programming language created by Dennis Ritchie at Bell Labs in 1972. Key features include:

  • Simple: Minimal set of keywords and straightforward syntax
  • Portable: Code can run on different platforms with minimal changes
  • Efficient: Produces fast and optimized code
  • Low-level Access: Direct memory manipulation via pointers
  • Structured: Supports structured programming constructs
  • Extensible: Can be extended with user-defined functions and types
  • Foundation: Basis for many modern languages (C++, Java, Python)

Q5: What is the difference between %d and %i in printf?

In printf, both %d and %i are identical - they both print integers.

In scanf, %d expects a decimal integer, while %i can accept decimal, octal (with 0 prefix), or hexadecimal (with 0x prefix).

#include 

int main() {
    int num1, num2;
    
    // %d - decimal only
    scanf("%d", &num1);
    
    // %i - decimal, octal, or hexadecimal
    scanf("%i", &num2);
    
    printf("num1: %d, num2: %d\n", num1, num2);
    
    return 0;
}
Q6: What is the difference between ++i and i++?

++i (Pre-increment): Increments the value first, then uses it.

i++ (Post-increment): Uses the value first, then increments it.

#include 

int main() {
    int i = 5;
    int j;
    
    // Pre-increment
    j = ++i; // i becomes 6, j becomes 6
    printf("i = %d, j = %d\n", i, j); // i = 6, j = 6
    
    // Reset
    i = 5;
    
    // Post-increment
    j = i++; // j becomes 5, i becomes 6
    printf("i = %d, j = %d\n", i, j); // i = 6, j = 5
    
    return 0;
}

4. Functions

Q9: What is the difference between call by value and call by reference?
Aspect Call by Value Call by Reference
Parameter Value is copied Address is passed
Modification Original not affected Original is affected
Memory Separate memory for copy Same memory accessed
Performance Slower for large data Faster (no copying) 
#include 

// Call by value
void swapByValue(int a, int b) {
    int temp = a;
    a = b;
    b = temp;
    // Original values unchanged
}

// Call by reference
void swapByReference(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
    // Original values changed
}

int main() {
    int x = 10, y = 20;
    
    swapByValue(x, y);
    printf("After swapByValue: x=%d, y=%d\n", x, y); // x=10, y=20
    
    swapByReference(&x, &y);
    printf("After swapByReference: x=%d, y=%d\n", x, y); // x=20, y=10
    
    return 0;
}
Q10: What is recursion in C?

Recursion is when a function calls itself to solve a problem. Each recursive call works on a smaller instance of the problem.

#include 

// Factorial using recursion
int factorial(int n) {
    if (n <= 1) { // Base case
        return 1;
    }
    return n * factorial(n - 1); // Recursive case
}

// Fibonacci using recursion
int fibonacci(int n) {
    if (n <= 1) {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

int main() {
    printf("Factorial of 5: %d\n", factorial(5)); // 120
    printf("Fibonacci of 10: %d\n", fibonacci(10)); // 55
    
    return 0;
}
Tip: Always have a base case to prevent infinite recursion. Recursion can lead to stack overflow if not properly handled.

Q12: What is a null-terminated string in C?

In C, strings are arrays of characters terminated by a null character ('\0'). The null character marks the end of the string.

#include 
#include 

int main() {
    // String literal (null-terminated)
    char str1[] = "Hello";
    // H e l l o \0
    
    // Manual string
    char str2[] = {'H', 'i', '\0'};
    
    // Without null terminator - undefined behavior
    char str3[] = {'H', 'e', 'l', 'l', 'o'}; // Not a valid string
    
    printf("str1: %s\n", str1); // Hello
    printf("str2: %s\n", str2); // Hi
    printf("Length of str1: %zu\n", strlen(str1)); // 5
    
    return 0;
}
Important: Always ensure strings are null-terminated. Functions like strlen, strcpy, etc. rely on the null terminator.

6. Pointers

Q13: What is a pointer and how is it used?

A pointer is a variable that stores the memory address of another variable. Pointers are used for:

  • Direct memory access
  • Dynamic memory allocation
  • Pass by reference
  • Data structures (linked lists, trees)
  • Efficient array handling
#include 

int main() {
    int num = 42;
    int *ptr;      // Pointer declaration
    ptr = #    // Store address of num
    
    printf("Value of num: %d\n", num);        // 42
    printf("Address of num: %p\n", &num);     // Address
    printf("Value of ptr: %p\n", ptr);        // Address
    printf("Value pointed by ptr: %d\n", *ptr); // 42
    
    *ptr = 100;    // Modify value through pointer
    printf("New value of num: %d\n", num);    // 100
    
    return 0;
}
Q14: What is the difference between *ptr and &ptr?

*ptr (Dereference operator): Gets the value stored at the address pointed to by ptr.

&ptr (Address-of operator): Gets the memory address of the variable ptr.

#include 

int main() {
    int num = 10;
    int *ptr = #
    
    printf("num: %d\n", num);        // 10
    printf("*ptr: %d\n", *ptr);      // 10 (value at address)
    printf("&num: %p\n", &num);      // Address of num
    printf("ptr: %p\n", ptr);        // Address stored in ptr
    printf("&ptr: %p\n", &ptr);      // Address of ptr variable
    
    return 0;
}
Q15: What is pointer arithmetic?

Pointer arithmetic allows you to perform arithmetic operations on pointers. The operations are scaled by the size of the data type.

#include 

int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int *ptr = arr;
    
    printf("*ptr: %d\n", *ptr);      // 10
    printf("*(ptr+1): %d\n", *(ptr+1)); // 20
    printf("*(ptr+2): %d\n", *(ptr+2)); // 30
    
    ptr++; // Move to next integer (adds 4 bytes typically)
    printf("After ptr++: %d\n", *ptr); // 20
    
    ptr += 2; // Move ahead by 2 integers
    printf("After ptr+=2: %d\n", *ptr); // 40
    
    // Difference between pointers
    int *ptr1 = &arr[0];
    int *ptr2 = &arr[3];
    printf("Difference: %ld\n", ptr2 - ptr1); // 3
    
    return 0;
}

7. Memory Management

Q16: What is dynamic memory allocation in C?

Dynamic memory allocation allows you to allocate memory at runtime using functions from stdlib.h:

  • malloc: Allocates uninitialized memory
  • calloc: Allocates and initializes memory to zero
  • realloc: Reallocates memory to new size
  • free: Deallocates previously allocated memory
#include 
#include 

int main() {
    // malloc - uninitialized
    int *ptr1 = (int*)malloc(5 * sizeof(int));
    if (ptr1 == NULL) {
        printf("Memory allocation failed\n");
        return 1;
    }
    
    // calloc - initialized to zero
    int *ptr2 = (int*)calloc(5, sizeof(int));
    
    // realloc - resize
    ptr1 = (int*)realloc(ptr1, 10 * sizeof(int));
    
    // free - deallocate
    free(ptr1);
    free(ptr2);
    
    return 0;
}
Important: Always check if malloc/calloc returns NULL (allocation failed). Always free allocated memory to prevent memory leaks.
Q17: What is the difference between malloc and calloc?
Aspect malloc calloc
Initialization Uninitialized (garbage values) Initialized to zero
Parameters Size in bytes Number of elements, size of each
Speed Faster Slower (due to initialization)
Return void* void* 
#include 
#include 

int main() {
    int *ptr1 = (int*)malloc(5 * sizeof(int));
    int *ptr2 = (int*)calloc(5, sizeof(int));
    
    // malloc - values are garbage
    for (int i = 0; i < 5; i++) {
        printf("malloc[%d]: %d\n", i, ptr1[i]);
    }
    
    // calloc - values are zero
    for (int i = 0; i < 5; i++) {
        printf("calloc[%d]: %d\n", i, ptr2[i]);
    }
    
    free(ptr1);
    free(ptr2);
    
    return 0;
}

8. Structures and Unions

Q18: What is the difference between struct and union?
Aspect struct union
Memory Sum of all members Size of largest member
Access All members accessible simultaneously Only one member accessible at a time
Initialization All members can be initialized Only first member can be initialized
Use Case Store different data types together Store one of several types 
#include 

struct Student {
    int id;
    char name[20];
    float marks;
};

union Data {
    int i;
    float f;
    char str[20];
};

int main() {
    struct Student s = {1, "John", 85.5};
    printf("Struct size: %zu\n", sizeof(s)); // Sum of all members
    
    union Data d;
    d.i = 10;
    printf("Union size: %zu\n", sizeof(d)); // Size of largest member
    printf("d.i: %d\n", d.i); // 10
    
    d.f = 3.14;
    printf("d.f: %f\n", d.f); // 3.14
    printf("d.i: %d\n", d.i); // Undefined (corrupted)
    
    return 0;
}
Q19: What is a typedef in C?

typedef is used to create a new name (alias) for an existing data type. It improves code readability and makes it easier to change types later.

    int y;

9. File Handling

Q20: What are the different file modes in C?
Mode Description
"r" Read - opens file for reading (file must exist)
"w" Write - creates new file or truncates existing
"a" Append - opens file for appending (creates if not exists)
"r+" Read/Write - opens for reading and writing (file must exist)
"w+" Read/Write - creates or truncates file
"a+" Read/Append - opens for reading and appending 
#include 

int main() {
    FILE *file;
    
    // Write to file
    file = fopen("test.txt", "w");
    if (file == NULL) {
        printf("Error opening file\n");
        return 1;
    }
    
    fprintf(file, "Hello, World!\n");
    fputs("This is C programming.\n", file);
    fclose(file);
    
    // Read from file
    file = fopen("test.txt", "r");
    if (file == NULL) {
        printf("Error opening file\n");
        return 1;
    }
    
    char buffer[100];
    while (fgets(buffer, sizeof(buffer), file) != NULL) {
        printf("%s", buffer);
    }
    
    fclose(file);
    
    return 0;
}
